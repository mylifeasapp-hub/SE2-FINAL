    public static boolean isNotRegistered(String eMail) throws DatabaseException, SQLException {
        int anzahl = 0;
        String sql = "SELECT Count(*) FROM carpool.reg_user WHERE carpool.reg_user.reg_user_email = ? ";
        PreparedStatement statement = JDBCConnection.getInstance().getPreparedStatement(sql);
        statement.setString(1, eMail);

        try(ResultSet rs = statement.executeQuery()) {
            while (rs.next()) {
                anzahl = rs.getInt(1);
            }
        }
        if(anzahl > 0){
            return false;
        }
        else{
            return true;
        }
    }

    public static String getGender(String regAnrede){
        if(regAnrede.matches("Frau")){
            return "f";
        }
        else if(regAnrede.matches("Herr")){
            return "m";
        }
        else{
            return "d";
        }
    }

    public static String getGenderFullname(String regAnrede){
        if(regAnrede.matches("f")){
            return "Frau";
        }
        else if(regAnrede.matches("m")){
            return "Herr";
        }
        else{
            return "-";
        }
    }
